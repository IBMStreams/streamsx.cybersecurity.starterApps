 list<IPv6> convertIPV6AddressStringsToNumerics(list<rstring> addresses)
 {
 	mutable list<IPv6> ipList = [ ] ;
 	for(rstring ip in addresses)
 	{
 		appendM(ipList, convertIPV6AddressStringToNumeric(ip)) ;
 	}
 
 	return ipList ;
 }
